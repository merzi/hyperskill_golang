Implementing else if

At what value of number would this Go program print the message within the else if block?

package main

import "fmt"

func main() {
    var number int
    fmt.Scanf("%d", &number)

    if number > 123 && number < 567 {
        fmt.Println("Do not print me!")
    } else if number > 567 {
        fmt.Println("Print me!!!")
    } else {
        fmt.Println("Do not print me either!")
    }
}
